def plot_mean(all_reward,num_statistics):
    mean_reward=[]
    for i in range( len(all_reward)-num_statistics ):
        mean_reward.append( np.mean(all_reward[i:i+num_statistics]) )

    max_reward=[]
    for i in range( len(all_reward)-num_statistics ):
        max_reward.append( max(all_reward[i:i+num_statistics]) )

    min_reward=[]
    for i in range( len(all_reward)-num_statistics ):
        min_reward.append( min(all_reward[i:i+num_statistics]) )

    fig = plt.figure()
    ax = fig.add_subplot(111)
    ax.plot(range(len(mean_reward)), mean_reward, color='black', linewidth=1) # plot line
    ax.plot(range(len(max_reward)), max_reward, color='red', linewidth=1) # plot line
    ax.plot(range(len(min_reward)), min_reward, color='blue', linewidth=1) # plot line
    # ax.set_ylim(-5, 1)
    # ax.set_xlim(0, 3500)
    plt.grid(True)
    plt.show()

import tensorflow as tf
import numpy as np

# fully connect classification
class FC_Classify():
    def __init__(self):
        self.history_length = 4
        self.state_dim = 4+2*self.history_length
        self.action_dim = 4
        
        self.state_input = tf.placeholder("float",[None,self.state_dim])
        self.action_label = tf.placeholder("float",[None,self.action_dim])
        
        self.action_output = self.build_fc_classify_networks(self.state_input)
        
        self.train, self.cost = self.create_train_opt()
        
        # Init session
        self.session = tf.InteractiveSession()
        self.session.run(tf.global_variables_initializer())
        
    def build_fc_classify_networks(self,state_input):
        # args:
        self.action_dim
        
        h_layer01 = tf.layers.dense( state_input, 200, activation=tf.nn.relu )
        h_layer02 = tf.layers.dense( h_layer01, CONFIG["hidd_nodes"], activation=tf.nn.relu )
        if CONFIG["hidd_layers"]==2:
            return tf.nn.softmax( tf.layers.dense( h_layer02, self.action_dim ) , name="prob")
        elif CONFIG["hidd_layers"]==3:
            h_layer03 = tf.layers.dense( h_layer02, 60, activation=tf.nn.relu )
            return tf.nn.softmax( tf.layers.dense( h_layer03, self.action_dim ) , name="prob")
    
    def create_train_opt(self):
        cost = tf.reduce_sum((self.action_output - self.action_label) ** 2)
        train = tf.train.GradientDescentOptimizer(0.0001).minimize(cost)
        return train, cost
